  "@type": "MedicalOrganization",
  "name": "Colorado Recovery Center",
  "description": "Addiction treatment and sober living in Colorado",
  "url": "https://example.com",
  "telephone": "+1-555-123-4567",
  "address": {
    "@type": "PostalAddress",
    "streetAddress": "123 Recovery St",
    "addressLocality": "Denver",
    "addressRegion": "CO",
